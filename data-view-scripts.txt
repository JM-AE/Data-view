from tkinter import ttk

class DataDashboard:
    def __init__(self, root):
        self.root = root
        self.root.title("Data Handling Dashboard")
        self.data = None
        self.plot_frame = None  # Frame to embed the plot
        self.visualization_options = {  # Available visualization functions
            "Table": self.display_table,
            "Histogram": self.create_histogram,
            "Correlation Matrix": self.create_correlation_matrix,
            "Pie Chart": self.create_pie_chart
        }

        # Create main frames
        self.create_buttons()
        self.create_display_area()

    def create_buttons(self):
        button_frame = tk.Frame(self.root)
        button_frame.pack(side=tk.TOP, fill=tk.X)

        tk.Button(button_frame, text="Load Data", command=self.load_data).pack(side=tk.LEFT, padx=5, pady=5)
        tk.Button(button_frame, text="Describe Data", command=self.describe_data).pack(side=tk.LEFT, padx=5, pady=5)

        # Visualization Dropdown
        self.selected_option = tk.StringVar(self.root)
        self.selected_option.set("Select Visualization")
        dropdown = ttk.OptionMenu(
            button_frame, self.selected_option, *self.visualization_options.keys(), command=self.visualize_data
        )
        dropdown.pack(side=tk.LEFT, padx=5, pady=5)

        tk.Button(button_frame, text="Manipulate Data", command=self.manipulate_data).pack(side=tk.LEFT, padx=5, pady=5)

    def create_display_area(self):
        self.text_area = tk.Text(self.root, wrap=tk.WORD, height=20, width=100)
        self.text_area.pack(side=tk.BOTTOM, fill=tk.BOTH, expand=True, padx=5, pady=5)

    def load_data(self):
        file_path = filedialog.askopenfilename(filetypes=[("CSV Files", "*.csv"), ("Excel Files", "*.xlsx")])
        if file_path:
            try:
                if file_path.endswith(".csv"):
                    self.data = pd.read_csv(file_path)
                elif file_path.endswith(".xlsx"):
                    self.data = pd.read_excel(file_path)
                self.text_area.delete(1.0, tk.END)
                self.text_area.insert(tk.END, "Data loaded successfully.\n")
                self.text_area.insert(tk.END, f"Data preview:\n{self.data.head()}\n")
            except Exception as e:
                messagebox.showerror("Error", f"Failed to load data: {str(e)}")

    def describe_data(self):
        if self.data is not None:
            self.text_area.delete(1.0, tk.END)
            self.text_area.insert(tk.END, f"Data Description:\n{self.data.describe()}\n")
        else:
            messagebox.showwarning("No Data", "Please load data first.")

    def visualize_data(self, selected_option):
        if self.data is not None:
            plot_function = self.visualization_options.get(selected_option)
            if plot_function:
                if selected_option in ["Histogram", "Pie Chart"]:
                    column = tk.simpledialog.askstring("Column", "Enter column name:")
                    if column in self.data.columns:
                        self.display_plot(plot_function(column))
                    else:
                        messagebox.showerror("Error", f"Column '{column}' not found in data.")
                else:
                    self.display_plot(plot_function())
        else:
            messagebox.showwarning("No Data", "Please load data first.")

    def manipulate_data(self):
        if self.data is not None:
            options = ["Drop NA", "Fill NA"]
            option = tk.simpledialog.askstring("Manipulation Options", f"Choose one: {', '.join(options)}")
            if option == "Drop NA":
                self.data = self.data.dropna()
                messagebox.showinfo("Manipulation", "Missing values dropped.")
            elif option == "Fill NA":
                fill_value = tk.simpledialog.askstring("Fill Value", "Enter value to fill missing data:")
                self.data = self.data.fillna(fill_value)
                messagebox.showinfo("Manipulation", "Missing values filled.")
        else:
            messagebox.showwarning("No Data", "Please load data first.")

    def display_table(self):
        self.text_area.delete(1.0, tk.END)
        self.text_area.insert(tk.END, f"Data Table:\n{self.data.head()}\n")
        return None  # No plot to display

    def create_histogram(self, column):
        fig = Figure(figsize=(6, 4))
        ax = fig.add_subplot(111)
        ax.hist(self.data[column], bins=20, color='blue', edgecolor='black')
        ax.set_title(f"Histogram of {column}")
        ax.set_xlabel(column)
        ax.set_ylabel("Frequency")
        return fig

    def create_correlation_matrix(self):
        fig = Figure(figsize=(6, 4))
        ax = fig.add_subplot(111)
        sns.heatmap(self.data.corr(), annot=True, cmap="coolwarm", ax=ax)
        ax.set_title("Correlation Matrix")
        return fig

    def create_pie_chart(self, column):
        fig = Figure(figsize=(6, 4))
        ax = fig.add_subplot(111)
        self.data[column].value_counts().plot.pie(autopct='%1.1f%%', ax=ax)
        ax.set_title(f"Pie Chart of {column}")
        ax.set_ylabel("")  # Remove default ylabel
        return fig

    def display_plot(self, fig):
        if self.plot_frame:
            self.plot_frame.destroy()  # Clear previous plot

        self.plot_frame = tk.Frame(self.root)
        self.plot_frame.pack(side=tk.TOP, fill=tk.BOTH, expand=True)

        if fig:  # Only embed plot if a figure is provided
            canvas = FigureCanvasTkAgg(fig, master=self.plot_frame)
            canvas.draw()
            canvas.get_tk_widget().pack(side=tk.TOP, fill=tk.BOTH, expand=True)
